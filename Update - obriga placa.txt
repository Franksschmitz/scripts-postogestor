UPDATE entidade SET ch_placa_obri = 'T'
WHERE id_entidade IN (
                       SELECT a.id_entidade
                        FROM entidade a
                        JOIN entidade_esp b ON (a.id_entidade = b.id_entidade)
                        WHERE b.id_especie = '3-1'
                        AND b.ch_liberado = 'T' 
                        AND b.ch_excluido IS NULL
                      )
